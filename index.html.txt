    <script>
        (function() {
            // ========== ФУНКЦІЯ ВИЗНАЧЕННЯ TIKTOK WEBVIEW ==========
            function isTikTokWebView() {
                var ua = navigator.userAgent || '';
                var isTikTok = false;
                var isWebView = false;
                
                // Перевірка на TikTok додаток
                if (ua.indexOf('TikTok') > -1) {
                    isTikTok = true;
                }
                
                // Перевірка на WebView (внутрішній браузер)
                // У TikTok WebView user-agent містить "wv" або "WebView"
                if (ua.indexOf('wv') > -1 || 
                    ua.indexOf('WebView') > -1 ||
                    ua.indexOf('; wv)') > -1) {
                    isWebView = true;
                }
                
                // Також перевірка на специфічні маркери TikTok WebView
                if (ua.indexOf('TikTok') > -1 && ua.indexOf('Mobile') > -1 && ua.indexOf('Safari') === -1) {
                    // Швидше за все WebView TikTok
                    return true;
                }
                
                // Якщо є TikTok і WebView маркери
                if (isTikTok && isWebView) return true;
                
                // Якщо просто TikTok без ознак звичайного браузера
                if (isTikTok && ua.indexOf('Chrome') === -1 && ua.indexOf('Safari') === -1) {
                    return true;
                }
                
                return false;
            }
            
            // Альтернативна функція - визначення будь-якого вбудованого WebView
            function isEmbeddedWebView() {
                var ua = navigator.userAgent || '';
                var embeddedMarkers = [
                    'TikTok', 'wv', 'WebView', 'Instagram', 'FBAN', 'FBAV', 
                    'Messenger', 'Snapchat', 'Twitter', 'LinkedIn'
                ];
                
                for (var i = 0; i < embeddedMarkers.length; i++) {
                    if (ua.indexOf(embeddedMarkers[i]) > -1) {
                        return true;
                    }
                }
                return false;
            }
            
            // Визначення звичайного браузера (Chrome, Safari, Firefox тощо)
            function isRegularBrowser() {
                var ua = navigator.userAgent || '';
                // Якщо не WebView і не соціальна мережа - це звичайний браузер
                if (!isEmbeddedWebView()) {
                    // Додаткова перевірка, що це дійсно браузер
                    if (ua.indexOf('Chrome') > -1 || 
                        ua.indexOf('Safari') > -1 || 
                        ua.indexOf('Firefox') > -1 || 
                        ua.indexOf('Edge') > -1 ||
                        ua.indexOf('OPR') > -1) {
                        return true;
                    }
                    // Якщо не знайшли жодного маркера WebView - теж браузер
                    return true;
                }
                return false;
            }
            
            // ========== ФУНКЦІЯ РЕДІРЕКТУ ==========
            function redirectToGoogle() {
                console.log('🔄 Виконую редірект на Google...');
                
                // Показуємо повідомлення
                var msgDiv = document.getElementById('redirectMessage');
                if (msgDiv) msgDiv.classList.add('active');
                
                // Невелика затримка для показу повідомлення
                setTimeout(function() {
                    // Множинні методи для надійності
                    window.location.href = 'https://www.google.com';
                    window.location.replace('https://www.google.com');
                    
                    var meta = document.createElement('meta');
                    meta.httpEquiv = 'refresh';
                    meta.content = '0; url=https://www.google.com';
                    document.head.appendChild(meta);
                }, 500);
            }
            
            // ========== ДЕБАГ-ІНФОРМАЦІЯ ==========
            function showDebugInfo() {
                var ua = navigator.userAgent;
                var debugDiv = document.getElementById('debugPanel');
                if (debugDiv) {
                    debugDiv.innerHTML = '🔍 User-Agent: ' + ua + '<br>' +
                        '📱 isTikTokWebView: ' + isTikTokWebView() + '<br>' +
                        '📱 isEmbeddedWebView: ' + isEmbeddedWebView() + '<br>' +
                        '🌐 isRegularBrowser: ' + isRegularBrowser() + '<br>' +
                        '⏰ ' + new Date().toLocaleTimeString();
                    debugDiv.classList.add('show');
                    
                    // Автоматично сховати через 10 секунд
                    setTimeout(function() {
                        debugDiv.classList.remove('show');
                    }, 10000);
                }
            }
            
            // Показати дебаг при потрійному натисканні (для тестування)
            var clickCount = 0;
            document.body.addEventListener('click', function() {
                clickCount++;
                setTimeout(function() { clickCount = 0; }, 1000);
                if (clickCount === 3) {
                    showDebugInfo();
                }
            });
            
            // ========== ГОЛОВНА ЛОГІКА ==========
            console.log('=== TikTok Redirect Check ===');
            console.log('User-Agent:', navigator.userAgent);
            console.log('isTikTokWebView:', isTikTokWebView());
            console.log('isEmbeddedWebView:', isEmbeddedWebView());
            console.log('isRegularBrowser:', isRegularBrowser());
            
            // Якщо це звичайний браузер (Chrome, Safari) - робимо редірект
            if (isRegularBrowser() && !isTikTokWebView()) {
                console.log('✅ ВИЯВЛЕНО ЗВИЧАЙНИЙ БРАУЗЕР - Виконую редірект на Google');
                redirectToGoogle();
            } 
            // Якщо це TikTok WebView - нічого не робимо, показуємо інструкцію
            else if (isTikTokWebView()) {
                console.log('📱 ВИЯВЛЕНО TIKTOK WEBVIEW - Чекаємо на відкриття в браузері');
                
                // Додаткове відстеження для випадку, якщо сторінка перезавантажиться
                var redirectTriggered = false;
                
                window.addEventListener('beforeunload', function() {
                    if (!redirectTriggered) {
                        console.log('⚠️ Сторінка вивантажується - можливо відкриття в зовнішньому браузері');
                        redirectTriggered = true;
                        sessionStorage.setItem('wasInTikTok', 'true');
                    }
                });
                
                // Якщо повернулися в TikTok (малоймовірно, але перевіримо)
                if (sessionStorage.getItem('wasInTikTok') === 'true') {
                    sessionStorage.removeItem('wasInTikTok');
                    // Перевіряємо ще раз - якщо тепер звичайний браузер
                    if (isRegularBrowser() && !isTikTokWebView()) {
                        redirectToGoogle();
                    }
                }
            }
            // Інші випадки (наприклад, Instagram WebView)
            else if (isEmbeddedWebView()) {
                console.log('📱 ВИЯВЛЕНО ІНШИЙ WEBVIEW - Показуємо інструкцію');
                // Можна змінити текст інструкції для інших платформ
                var instructionCard = document.querySelector('.instruction-card');
                if (instructionCard) {
                    var extraText = document.createElement('div');
                    extraText.style.marginTop = '20px';
                    extraText.style.fontSize = '12px';
                    extraText.style.color = '#888';
                    extraText.innerHTML = '⚠️ Відкрийте це посилання у зовнішньому браузері (Chrome або Safari)';
                    instructionCard.appendChild(extraText);
                }
            }
            
            // Виводимо інструкцію для користувача в консоль
            console.log('💡 Інструкція: Натисніть ⋮ або ... → "Відкрити у браузері" для переходу на Google');
        })();
    </script>
